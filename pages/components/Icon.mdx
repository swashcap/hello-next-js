## Usage

Import the `Icon` component into a React component in your project:

```typescript
import {Icon} from 'hello-next-js'
```

Use the `Icon` component in a component:

```tsx
import React from 'react'
import {Icon} from 'hello-next-js'

export const MyComponent: React.FC<any> = () => {
  return (
    <div>
      <button className="my-button">
        Cancel
        <Icon name="close" />
      </button>
    </div>
  )
}
```
