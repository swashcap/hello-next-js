#Hello, Next.js!

This tests Next.js with prism-react-renderer. Here's a sample code block:

import React from 'react';
export interface Props extends React.HTMLAttributes<HTMLDivElement> {
  /**
   * Whether the component is a sample.
   *
   * @default false
   */
  isSample?: boolean
}
export const SampleCompnent: React.FC<Props> = (props) => {
  const {
    isSample,
    ...rest
  } = props;
  return (
    <div {...rest}>
      {isSample ? 'It\'s a sample' : 'No sample here'}
    </div>
  );
};
